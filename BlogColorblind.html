#######################################################################
#This is an example of how to use the package colorBlindness          #
#which runs your graphic through the different forms of colorblindness#
#and also provides a desaturated (greyscale printed) version.         #
#######################################################################
#install.packages("colorBlindness")
library(colorBlindness)
## Warning: package 'colorBlindness' was built under R version 4.4.1
#https://cran.r-project.org/web/packages/colorBlindness/vignettes/colorBlindness.html
#This is very easy to do with ggplot objects
library(ggplot2)
#create a ggplot
plot1 <- ggplot(data = iris, aes(x = Sepal.Length, fill = Species))+
  geom_histogram()
#create a simulation of the colorblindness
cvdPlot(plot1)
## `stat_bin()` using `bins = 30`. Pick better value with `binwidth`.

#It is harder to do with non-ggplot items. You have to export them as pngs.
#Then reimport them and turn them into rasters
library(grid)
library(png)

#create and export the png
png(filename = "plot2.png")
plot(iris$Sepal.Length~iris$Sepal.Width, 
       col = ifelse(iris$Species == "versicolor", "red", "green"),
     pch=16)
dev.off()
## png 
##   2
#reimport it, turn into raster
img <- readPNG("plot2.png")
g <- rasterGrob(img)

#now, simulate
cvdPlot(g)
